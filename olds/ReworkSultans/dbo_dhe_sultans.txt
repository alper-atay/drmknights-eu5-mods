# dbo_history_yavuz_sultan_selim_born
dbo_dhe_sultans.301 = {
	type = country_event
	title = dbo_dhe_sultans.301.title
	desc = dbo_dhe_sultans.301.desc
	historical_info = dbo_dhe_sultans.301.historical_info
	image = "gfx/interface/illustrations/event/backgrounds/dbo_history_yavuz_sultan_selim_born.dds"
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1470.1.1
		to = 9999.1.1
		monthly_chance = 100
	}
	trigger = {
		exists = character:dbo_fatih_sultan_mehmed
		OR = {
			AND = {
				has_ruler = yes
				ruler = {
					father = character:dbo_fatih_sultan_mehmed
				}
			}
			AND = {
				has_heir = yes
				heir = {
					father = character:dbo_fatih_sultan_mehmed
				}
			}
		}
	}
	immediate = {
		if = {
			limit = {
				has_ruler = yes
				ruler = {
					father = character:dbo_fatih_sultan_mehmed
				}
			}
			ruler = {
				save_temporary_scope_as = target_father
			}
		}
		else_if = {
			limit = {
				has_heir = yes
				heir = {
					father = character:dbo_fatih_sultan_mehmed
				}
			}
			heir = {
				save_temporary_scope_as = target_father
			}
		}
		else = {
			create_character = {
				first_name = Bayezid
				father = character:dbo_fatih_sultan_mehmed
				birth_location = root.capital
				age = {
					min = 16
					max = 30
				}
				culture = root.culture
				religion = root.religion
				adm = {
					60
					80
				}
				dip = {
					60
					80
				}
				mil = {
					60
					80
				}
				save_temporary_scope_as = target_father
			}
		}
		create_character = {
			first_name = Selim
			nickname = nick_Yavuz
			father = scope:target_father
			birth_location = root.capital
			age = 0
			culture = root.culture
			religion = root.religion
			fertility = 100
			add_trait = trait:child_prodigy
			script = dbo_yavuz_sultan_selim
		}
		character:dbo_yavuz_sultan_selim = {
			add_character_modifier = {
				modifier = blocked_from_being_ruler
				years = modifier_duration_permanent
				mode = add_and_extend
			}
		}
	}
	option = {
		name = dbo_dhe_sultans.301.a
		historical_option = yes
		dbo_event_mild_bonus = yes
	}
}

		exists = character:dbo_kanuni_sultan_suleiman

			is_alive = yes
			is_adult = yes
		}
	}
	immediate = {
		kill_character = {
			target = character:dbo_yavuz_sultan_selim
			reason = sudden_illness
		}
	}
	option = {
		name = dbo_dhe_sultans.307.a
		historical_option = yes
	}
}
