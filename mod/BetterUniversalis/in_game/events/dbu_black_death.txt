# dbu_black_death_ended
dbu_black_death.100 = {
	type = country_event
	title = dbu_black_death.100.title
	desc = dbu_black_death.100.desc
	trigger = {
		situation:black_death = {
			situation_is_active = no
			situation_has_ended = yes
		}
		has_variable = had_black_death
		NOT = {
			has_variable = dbu_had_black_death_recovery
		}
	}
	option = {
		name = dbu_black_death.100.a
		add_country_modifier = {
			modifier = dbu_recovery_after_the_black_death
			years = modifier_duration_years_century
			mode = add_and_extend
		}
		add_country_modifier = {
			modifier = innovation_wave
			years = modifier_duration_years_long
			mode = extend
		}
		add_gold = {
			value = country_tax_base
			multiply = 12
			add = 100
		}
		set_variable = {
			name = dbu_had_black_death_recovery
			value = yes
		}
	}
}
