namespace = dbo_religious_building
dbo_religious_building.1 = {
	type = country_event
	title = dbo_religious_building.1.title
	desc = dbo_religious_building.1.desc
	illustration_tags = {
		10 = regular
		10 = interior
	}
	trigger = {
		tag = TUR
		country_exists = c:TUR
		c:TUR = {
			religion.group = religion_group:muslim
			any_owned_non_rural_location = {
				has_building_with_at_least_one_level = temple
				has_building_with_at_least_one_level = madrasa
				NOT = {
					has_building = building_type:great_madrasa
				}
			}
		}
	}
	immediate = {
		event_illustration_estate_effect = {
			foreground = estate_type:burghers_estate
			background = estate_type:clergy_estate
		}
		c:TUR = {
			random_owned_non_rural_location = {
				limit = {
					has_building_with_at_least_one_level = temple
					has_building_with_at_least_one_level = madrasa
					NOT = {
						has_building = building_type:great_madrasa
					}
				}
				save_scope_as = madrassah_province
			}
		}
	}
	option = {
		name = dbo_religious_building.1.a
		scope:madrassah_province = {
			if = {
				limit = {
					NOT = {
						has_building = building_type:great_madrasa
					}
				}
				change_building_level_in_location = {
					building = building_type:great_madrasa
					value = 1
				}
			}
		}
	}
}
