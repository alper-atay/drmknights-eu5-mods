namespace = dbo_conquest

# dbo_conquest_anatolian_turks_claim
dbo_conquest.101 = {
    type = country_event
    category = situation_event
    title = dbo_conquest_common_title
    desc = dbo_conquest_common_desc
    immediate = {
        ruler_or_regent ?= {
            save_scope_as = target_character1
        }
        scope:target.owner = {
            save_scope_as = target_country
            ruler_or_regent ?= {
                save_scope_as = target_character2
            }
        }
    }
    option = {
        name = dbo_conquest_common_option
        add_casus_belli = {
            target = scope:target.owner
            type = casus_belli:dbo_cb_conquest_anatolian_turks
            province = scope:target
        }
        refresh_map_colors = yes
    }
}
