namespace = dbo_dhe_history_ottoman
dbo_dhe_history_ottoman.1 = {
	type = country_event
	title = dbo_dhe_history_ottoman.1.title
	desc = dbo_dhe_history_ottoman.1.desc
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1337.1.1
		to = 9999.1.1
		monthly_chance = 100
	}
	trigger = {
		NOT = {
			religion = religion:tengri
		}
	}
	option = {
		name = dbo_dhe_history_ottoman.1.a
		historical_option = yes
		change_religion = religion:tengri
		change_religion_for_ruler_and_family = {
			country = root
			religion = religion:tengri
		}
		set_liturgical_language = culture.dialect
		every_pop = {
			limit = {
				NOT = {
					religion = religion:tengri
				}
				culture = {
					has_culture_group = culture_group:turkic_group
				}
			}
			split_pop = {
				fraction = 0.4
				religion = religion:tengri
			}
		}
		every_owned_location = {
			construct_building = {
				building_type = building_type:dbo_otag
				instant = yes
			}
		}
		dbo_event_extreme_bonus = yes
		add_policy = policy:monogamous_marriage
		add_policy = policy:heir_same_religion
		add_policy = policy:balanced_court_policy
		unlock_policy_effect = {
			type = law_of_the_pious_emperor
		}
		add_policy = policy:law_of_the_pious_emperor
		add_gold = 1000
	}
}

# Ottoman Interregnum Begin
dbo_dhe_history_ottoman.101 = {
	type = country_event
	title = dbo_dhe_history_ottoman.101.title
	desc = dbo_dhe_history_ottoman.101.desc
	historical_info = dbo_dhe_history_ottoman.101.historical_info
	image = "gfx/interface/illustrations/event/backgrounds/dbo_history_ottoman_interregnum_begin.dds"
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1354.1.1
		to = 9999.1.1
		monthly_chance = 100
	}
	trigger = {
		exists = character:dbo_yildirim_sultan_bayezid
		character:dbo_yildirim_sultan_bayezid = {
			is_alive = no
		}
		any_character = {
			father = character:dbo_yildirim_sultan_bayezid
			is_female = no
			is_adult = yes
			count > 1
		}
	}
	option = {
		name = dbo_dhe_history_ottoman.101.a
		historical_option = yes
		dbo_event_extreme_penalty = yes
		add_country_modifier = {
			modifier = dbo_ottoman_interregnum
			years = 11
			mode = add_and_extend
		}
		custom_tooltip = {
			text = dbo_history_ottoman_interregnum_rebel_progress_gain_tt
			every_rebel = {
				add_rebel_progress = rebel_progress_severe_penalty
			}
		}
		custom_tooltip = {
			text = dbo_history_ottoman_interregnum_tt
			every_pop = {
				limit = {
					owner = root
				}
				add_pop_satisfaction = pop_satisfaction_severe_penalty
			}
		}
		set_variable = {
			name = dbo_ottoman_interregnum_end_year
			value = {
				value = current_year
				add = 10
			}
		}
	}
}

		from = 1354.1.1

		}
		custom_tooltip = {
			text = dbo_history_ottoman_interregnum_end_tt
			every_pop = {
				limit = {
					owner = root
				}

		}
		custom_tooltip = {

	}
