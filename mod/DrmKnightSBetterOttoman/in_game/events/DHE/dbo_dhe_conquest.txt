# dbo_conquest_anatolian_turks_stamp
dbo_dhe_conquest.1001 = {
	type = country_event
	title = dbo_dhe_conquest.1001.title
	desc = dbo_dhe_conquest.1001.desc
	historical_info = dbo_dhe_conquest.1001.historical_info
	image = "gfx/interface/illustrations/event/backgrounds/dbo_conquest_anatolian_turks.dds"
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1337.1.1
		to = 1837.1.1
		monthly_chance = 100
	}
	trigger = {
		NOT = {
			has_reform = government_reform:dbo_conquest_anatolian_turks_stamp
		}
		c:TUR = {
			dbo_conquest_anatolian_turks_check = yes
		}
	}
	option = {
		name = dbo_dhe_conquest.1001.a
		historical_option = yes
		set_variable = dbo_conquest_anatolian_turks_stamp_obtained
		add_reform = government_reform:dbo_conquest_anatolian_turks
		dbo_event_extreme_bonus = yes
	}
}
