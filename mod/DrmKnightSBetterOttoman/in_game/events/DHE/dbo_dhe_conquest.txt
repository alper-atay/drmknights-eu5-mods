namespace = dbo_dhe_conquest
dbo_dhe_conquest.1 = {
	type = country_event
	title = dbo_dhe_conquest.1.title
	desc = dbo_dhe_conquest.1.desc
	historical_info = dbo_dhe_conquest.1.historical_info
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1337.1.1
		to = 1837.1.1
		monthly_chance = 100
	}
	trigger = {
		NOT = {
			has_reform = government_reform:dbo_conquest_anatolian_turks
		}
		c:TUR = {
			dbo_conquest_anatolian_turks_achieved = yes
		}
	}
	option = {
		name = dbo_dhe_conquest.1.a
		set_variable = {
			name = dbo_conquest_anatolian_turks
			value = yes
		}
		historical_option = yes
		add_reform = government_reform:dbo_conquest_anatolian_turks
		add_legitimacy = legitimacy_extreme_bonus
		add_stability = stability_extreme_bonus
		add_prestige = prestige_extreme_bonus
	}
}
