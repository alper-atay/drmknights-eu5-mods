namespace = dbo_dhe_institution
dbo_dhe_institution.1 = {
	type = country_event
	title = dbo_dhe_institution.1.title
	desc = dbo_dhe_institution.1.desc
	image = "gfx/interface/illustrations/institutions/professional_armies.dds"
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1337.1.1
		to = 9999.1.1
		monthly_chance = 100
	}
	trigger = {
		knows_about_institution = institution:professional_armies
		NOT = {
			has_embraced_institution = institution:professional_armies
		}
	}
	option = {
		name = dbo_dhe_institution.1.a
		every_owned_location = {
			change_institution_progress = {
				type = institution:professional_armies
				value = institution_progress_extreme_bonus
			}
		}
		dbo_event_severe_bonus = yes
	}
}
