namespace = dbo_ottoman_empire
dbo_ottoman_empire.1 = {
	type = country_event
	title = dbo_ottoman_empire.1.title
	desc = dbo_ottoman_empire.1.desc
	historical_info = dbo_ottoman_empire.1.historical_info
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1337.1.1
		to = 1837.1.1
		monthly_chance = 25
	}
	trigger = {
		country_rank = country_rank:rank_empire
		NOT = {
			has_reform = government_reform:dbo_ottoman_empire
		}
		OR = {
			has_reform = government_reform:legacy_of_the_ottomans
			has_reform = government_reform:turkish_pashas
		}
	}
	option = {
		name = dbo_ottoman_empire.1.a
		historical_option = yes
		set_variable = {
			name = dbo_ottoman_empire
			value = yes
		}
		if = {
			limit = {
				has_reform = government_reform:legacy_of_the_ottomans
			}
			remove_reform = government_reform:legacy_of_the_ottomans
		}
		if = {
			limit = {
				has_reform = government_reform:turkish_pashas
			}
			remove_reform = government_reform:turkish_pashas
		}
		lock_government_reform_effect = {
			type = legacy_of_the_ottomans
		}
		lock_government_reform_effect = {
			type = turkish_pashas
		}
		add_reform = government_reform:dbo_ottoman_empire
		add_stability = stability_extreme_bonus
		add_legitimacy = legitimacy_extreme_bonus
	}
}
