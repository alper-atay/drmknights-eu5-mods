# Yildirim Bayezid Born
dbo_dhe_history.101 = {
	type = country_event
	title = dbo_dhe_history.101.title
	desc = dbo_dhe_history.101.desc
	historical_info = dbo_dhe_history.101.historical_info
	image = "gfx/interface/illustrations/event/backgrounds/dbo_history_yildirim_bayezid_born.dds"
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1351.1.1
		to = 1358.1.1
		monthly_chance = 100
	}
	trigger = {
		has_ruler = yes
		ruler = {
			is_adult = yes
		}
	}
	immediate = {
		create_character = {
			first_name = Bayezid
			nickname = nick_Yildirim
			dynasty = dynasty:ottomans_dynasty
			father = root.ruler
			birth_location = root.capital
			age = 0
			culture = culture:turkish_culture
			religion = religion:sunni
			fertility = 100
			adm = 5
			dip = 5
			mil = 5
			add_trait = trait:child_prodigy
			script = dbo_yildirim_bayezid
		}
	}
	option = {
		name = dbo_dhe_history.101.a
		set_variable = {
			name = dbo_var_yildirim_bayezid_case
			value = 1
		}
	}
}

	historical_info = dbo_dhe_history.102.historical_info
	image = "gfx/interface/illustrations/event/backgrounds/dbo_history_yildirim_bayezid_rise.dds"
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1370.1.1
		to = 1380.1.1
		monthly_chance = 100
	}
	trigger = {
		exists = character:dbo_yildirim_bayezid
		has_variable = dbo_var_yildirim_bayezid_case

		has_ruler = yes
		character:dbo_yildirim_bayezid = {
			is_alive = yes
			is_adult = yes
			is_ruler = no
		}
	}
	option = {
		name = dbo_dhe_history.102.a
		set_new_ruler = character:dbo_yildirim_bayezid
		character:dbo_yildirim_bayezid = {
			add_trait = trait:bold_fighter
			add_trait = trait:conqueror
			add_trait = trait:hussar_commander
		}
		change_variable = {
			name = dbo_var_yildirim_bayezid_case
			add = 1
		}
	}
}
