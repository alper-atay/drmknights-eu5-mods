# Yildirim Bayezid https://tr.wikipedia.org/wiki/I._Bayezid
dbo_dhe_history.1101 = {
    type = country_event
    title = dbo_dhe_history.1101.title
    desc = dbo_dhe_history.1101.desc
    historical_info = dbo_dhe_history.1101.historical_info
    image = "gfx/interface/illustrations/event/backgrounds/dbo_history_yildirim_bayezid_born.dds"
    fire_only_once = yes
    dynamic_historical_event = {
        tag = TUR
        from = 1351.1.1
        to = 1358.1.1
        monthly_chance = 100
    }
    trigger = {
        has_ruler = yes
        ruler ?= {
            is_adult = yes
        }
    }
    option = {
        name = dbo_dhe_history.1101.a
        create_character = {
            first_name = Bayezid
            nickname = nick_Yildirim
            dynasty = dynasty:ottomans_dynasty
            father = root.ruler
            birth_location = root.capital
            age = 0
            culture = culture:turkish_culture
            religion = religion:sunni
            fertility = 100
            adm = 5
            dip = 5
            mil = 5
            add_trait = trait:child_prodigy
            add_trait = trait:bold_fighter
            add_trait = trait:conqueror
            add_trait = trait:hussar_commander
            script = dbo_yildirim_bayezid
        }
    }
}
