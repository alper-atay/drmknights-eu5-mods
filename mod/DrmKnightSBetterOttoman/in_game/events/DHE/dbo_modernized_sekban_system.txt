		tag = TUR

		}
	}

		name = dbo_modernized_sekban_system.1.a
		set_variable = {
			name = dbo_modernized_sekban_system

		}
		historical_option = yes

			limit = {
				has_reform = government_reform:sekban_system

		}
		if = {
			limit = {
				has_variable = tur_sekban_system
			}

		}
		add_reform = government_reform:dbo_modernized_sekban_system
