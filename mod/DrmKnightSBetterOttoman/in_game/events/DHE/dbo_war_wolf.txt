namespace = dbo_war_wolf
dbo_war_wolf.1 = {
	type = country_event
	title = dbo_war_wolf.1.title
	desc = dbo_war_wolf.1.desc
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1337.1.1
		to = 1837.1.1
		monthly_chance = 25
	}
	trigger = {
		NOT = {
			has_reform = government_reform:dbo_war_wolf_bronze
		}
		num_locations >= 250
		num_of_non_rural >= 25
	}
	option = {
		name = dbo_war_wolf.1.a
		historical_option = yes
		set_variable = {
			name = dbo_war_wolf_bronze
			value = yes
		}
		add_reform = government_reform:dbo_war_wolf_bronze
		add_stability = stability_severe_bonus
	}
}
