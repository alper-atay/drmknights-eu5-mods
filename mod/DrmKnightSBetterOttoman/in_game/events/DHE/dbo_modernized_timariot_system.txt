namespace = dbo_modernized_timariot_system
dbo_modernized_timariot_system.1 = {
	type = country_event
	title = dbo_modernized_timariot_system.1.title
	desc = dbo_modernized_timariot_system.1.desc
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1337.1.1
		to = 1837.1.1
		monthly_chance = 25
	}
	trigger = {
		has_reform = government_reform:dbo_lessons_from_mongols
		NOT = {
			has_reform = government_reform:dbo_modernized_timariot_system
		}
		AND = {
			OR = {
				has_variable = expanded_timariot_system
				has_variable = abolished_timariot_system
			}
			has_advance = timariots
		}
	}
	option = {
		name = dbo_modernized_timariot_system.1.a
		if = {
			limit = {
				has_variable = abolished_timariot_system
			}
			remove_variable = abolished_timariot_system
		}
		set_variable = {
			name = dbo_modernized_timariot_system
			value = yes
		}
		historical_option = yes
		if = {
			limit = {
				has_reform = government_reform:basic_timariot_system
			}
			remove_reform = government_reform:basic_timariot_system
		}
		if = {
			limit = {
				has_reform = government_reform:expanded_timariot_system
			}
			remove_reform = government_reform:expanded_timariot_system
		}
		if = {
			limit = {
				has_variable = expanded_timariot_system
			}
			remove_variable = expanded_timariot_system
		}
		if = {
			limit = {
				has_reform = government_reform:dbo_lessons_from_mongols
			}
			remove_reform = government_reform:dbo_lessons_from_mongols
		}
		if = {
			limit = {
				has_variable = dbo_lessons_from_mongols
			}
			remove_variable = dbo_lessons_from_mongols
		}
		add_estate_satisfaction = {
			type = estate_type:nobles_estate
			value = estate_satisfaction_mild_penalty
		}
		add_estate_satisfaction = {
			type = estate_type:clergy_estate
			value = estate_satisfaction_mild_penalty
		}
		add_estate_satisfaction = {
			type = estate_type:peasants_estate
			value = estate_satisfaction_mild_penalty
		}
		add_stability = stability_extreme_penalty
		set_variable = {
			name = dbo_modernized_timariot_system
			value = yes
		}
		add_reform = government_reform:dbo_modernized_timariot_system
	}
}
