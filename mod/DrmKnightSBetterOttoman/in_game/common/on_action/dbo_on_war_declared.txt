#root = country, scope:actor = country who declares the war, scope:recipient = country who gets declared on, scope:war = war
on_war_declared = {
	on_actions = {
		dbo_on_conquest_with_cb
	}
}

dbo_on_conquest_with_cb = {
	trigger = {
		tag_exists = TUR
		scope:actor = {
			tag = TUR
		}
		scope:war = {
			has_casus_belli = yes
			casus_belli = casus_belli:dbo_cb_conquest_nile_river
		}
	}
	effect = {
		scope:recipient = {
			every_owned_location = {
				limit = {
					is_ownable = yes
				}
				add_location_modifier = {
					modifier = dbo_ottoman_scare
					years = 5
					mode = replace
				}
			}
		}
	}
}
