dbo_conquer_arabian_peninsula = {
    monthly_spawn_chance = monthly_spawn_chance_unique
    hint_tag = dbo_hint_conquer_arabian_peninsula
    can_start = {
        current_date > 1337.1.1
        situation:dbo_conquer_fertile_crescent = {
			situation_is_active = no
			situation_has_ended = yes
		}
    }
    can_end = {
        dbo_conquer_arabian_peninsula_achieved = yes
    }
    visible = {
        tag = TUR
    }
    on_start = {
        c:TUR ?= {
            trigger_event_non_silently = dbo_conquer_arabian_peninsula.1
        }
    }
    on_ended = {
        c:TUR ?= {
            trigger_event_non_silently = dbo_conquer_arabian_peninsula.2
        }
    }
}
