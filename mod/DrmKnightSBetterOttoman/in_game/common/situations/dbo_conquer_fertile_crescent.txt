dbo_conquer_fertile_crescent = {
    monthly_spawn_chance = monthly_spawn_chance_unique
    hint_tag = dbo_hint_conquer_fertile_crescent
    can_start = {
        current_date > 1337.1.1
    }
    can_end = {
        dbo_conquer_fertile_crescent_achieved = yes
    }
    visible = {
        tag = TUR
    }
    on_start = {
        c:TUR ?= {
            trigger_event_non_silently = dbo_conquer_fertile_crescent.1
        }
    }
    on_ended = {
        c:TUR ?= {
            trigger_event_non_silently = dbo_conquer_fertile_crescent.2
        }
    }
}
