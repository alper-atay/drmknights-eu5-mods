# dbo_history_kanuni_sultan_suleiman_born
dbo_dhe_sultans_kanuni_sultan_suleiman.1 = {
	type = country_event
	title = dbo_dhe_sultans_kanuni_sultan_suleiman.1.title
	desc = dbo_dhe_sultans_kanuni_sultan_suleiman.1.desc
	historical_info = dbo_dhe_sultans_kanuni_sultan_suleiman.1.historical_info
	image = "gfx/interface/illustrations/event/backgrounds/dbo_history_kanuni_sultan_suleiman_born.dds"
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1494.1.1
		to = 9999.1.1
		monthly_chance = 100
	}
	trigger = {
		exists = character:dbo_yavuz_sultan_selim
		character:dbo_yavuz_sultan_selim = {
			is_alive = yes
			is_adult = yes
			character_age > 24
		}
	}
	immediate = {
		character:dbo_yavuz_sultan_selim = {
			save_temporary_scope_as = target_father
		}
		if = {
			limit = {
				scope:target_father = {
					has_available_marriage_slot = yes
				}
			}
			dbo_create_and_marry_lady_with_save_temporary_scope_as = {
				married_character = scope:target_father
				save_temporary_scope_as = target_mother
			}
		}
		else = {
			random_character = {
				limit = {
					is_spouse_of = scope:target_father
					is_adult = yes
					age_in_years < 40
				}
				save_temporary_scope_as = target_mother
			}
		}
		create_character = {
			first_name = name_salomon
			nickname = nick_lawgiver
			father = scope:target_father
			mother = scope:target_mother
			birth_location = root.capital
			age = 0
			culture = root.culture
			religion = root.religion
			fertility = 100
			add_trait = trait:dbo_prince
			script = dbo_kanuni_sultan_suleiman
		}
		character:dbo_kanuni_sultan_suleiman = {
			set_child_education = child_education:expensive_in_depth_education
		}
	}
	option = {
		name = dbo_dhe_sultans_kanuni_sultan_suleiman.1.a
		historical_option = yes
		dbo_event_mild_bonus = yes
	}
}
