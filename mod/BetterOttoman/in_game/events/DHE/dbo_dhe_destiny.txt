dbo_dhe_destiny.302 = {
	title = dbo_dhe_destiny.302.title
	desc = dbo_dhe_destiny.302.desc
	image = "gfx/interface/illustrations/event/backgrounds/dbo_destiny_turkish_pashas_claim.dds"
	trigger = {
		has_variable = dbo_destiny_noble_privilege_pashas_rejected
	}
	option = {
		name = dbo_dhe_destiny.301.a
		historical_option = yes
		dbo_destiny_noble_turkish_pashas_grant_estate_privilege_effect = yes
		hidden_effect = {
			if = {
				limit = {
					has_variable = dbo_destiny_noble_privilege_pashas_rejected
				}
				remove_variable = dbo_destiny_noble_privilege_pashas_rejected
			}
		}
	}
	option = {
		name = dbo_dhe_destiny.302.b
		set_variable = {
			name = dbo_destiny_noble_privilege_pashas_rejected
			value = yes
		}
		add_estate_satisfaction = {
			type = estate_type:nobles_estate
			value = estate_satisfaction_severe_penalty
		}
		add_stability = stability_severe_penalty
		add_gold_to_estate = {
			estate_type = estate_type:nobles_estate
			value = {
				value = monthly_income_trade_and_tax
				multiply = 2
			}
		}
		custom_tooltip = {
			text = dbo_destiny_every_nobles_or_soldier_pops_lose_satisfaction_tt
			every_pop = {
				limit = {
					dbo_turkish_pashas_rebel_allegiance_trigger = yes
				}
				add_pop_satisfaction = pop_satisfaction_severe_penalty
			}
		}
		if = {
			limit = {
				any_rebel = {
					rebel_name_key = rebel_pashas
				}
			}
			every_rebel = {
				limit = {
					rebel_name_key = rebel_pashas
					rebel_progress > 0
				}
				add_rebel_progress = {
					value = rebel_progress
					add = rebel_progress_weak_penalty
					divide = 2
				}
			}
		}
		# 
		else = {
			create_rebel = {
				name = rebel_pashas
				category = estate
				estate = nobles_estate
				culture = root.culture
				save_temporary_scope_as = target_rebel
			}
			every_pop = {
				limit = {
					dbo_turkish_pashas_rebel_allegiance_trigger = yes
				}
				add_pop_satisfaction = pop_satisfaction_radical_penalty
				change_pop_allegiance = scope:target_rebel
			}
		}
		trigger_event_silently = {
			id = dbo_dhe_destiny.302
			months = 12
		}
	}
}
