namespace = dbo_dhe_sultans_fatih_sultan_mehmed
# dbo_history_fatih_sultan_mehmed_born
dbo_dhe_sultans_fatih_sultan_mehmed.1 = {
	type = country_event
	title = dbo_dhe_sultans_fatih_sultan_mehmed.1.title
	desc = dbo_dhe_sultans_fatih_sultan_mehmed.1.desc
	historical_info = dbo_dhe_sultans_fatih_sultan_mehmed.1.historical_info
	image = "gfx/interface/illustrations/event/backgrounds/dbo_history_fatih_sultan_mehmed_born.dds"
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1432.1.1
		to = 9999.1.1
		monthly_chance = 100
	}
	trigger = {
		OR = {
			AND = {
				has_ruler = yes
				ruler = {
					is_adult = yes
					character_age < 40
				}
			}
			AND = {
				has_heir = yes
				heir = {
					is_adult = yes
					character_age < 40
				}
			}
		}
	}
	immediate = {
		if = {
			limit = {
				has_heir = yes
				heir = {
					is_adult = yes
				}
			}
			heir = {
				save_temporary_scope_as = target_father
			}
		}
		else_if = {
			limit = {
				has_ruler = yes
				ruler = {
					is_adult = yes
				}
			}
			ruler = {
				save_temporary_scope_as = target_father
			}
		}
		if = {
			limit = {
				scope:target_father = {
					has_available_marriage_slot = yes
				}
			}
			dbo_create_and_marry_lady_with_save_temporary_scope_as = {
				married_character = scope:target_father
				save_temporary_scope_as = target_mother
			}
		}
		else = {
			random_character = {
				is_spouse_of = scope:target_father
				is_pregnant = no
				is_adult = yes
				age_in_years < 40
			}
			save_temporary_scope_as = target_mother
		}
		create_character = {
			first_name = name_muhammad
			nickname = nick_the_conquerer
			father = scope:target_father
			mother = scope:target_mother
			birth_location = root.capital
			age = 0
			culture = root.culture
			religion = root.religion
			fertility = 100
			add_trait = trait:dbo_prince
			script = dbo_fatih_sultan_mehmed
		}
		character:dbo_fatih_sultan_mehmed = {
			set_child_education = child_education:expensive_in_depth_education
		}
	}
	option = {
		name = dbo_dhe_sultans_fatih_sultan_mehmed.1.a
		historical_option = yes
		dbo_event_mild_bonus = yes
	}
}

dbo_dhe_sultans_fatih_sultan_mehmed.3 = {
	type = country_event

	image = "gfx/interface/illustrations/event/backgrounds/dbo_history_fatih_sultan_mehmed_ascend.dds"
	fire_only_once = yes
	dynamic_historical_event = {
		tag = TUR
		from = 1432.1.1
		to = 9999.1.1
		monthly_chance = 100
	}
	trigger = {
		exists = character:dbo_fatih_sultan_mehmed
		character:dbo_fatih_sultan_mehmed = {
			has_trait = dbo_sultan
			is_adult = yes
		}
		OR = {
			has_ruler = yes
			has_regent = yes
			character:dbo_fatih_sultan_mehmed = {
				is_ruler = yes
			}
		}
	}
	immediate = {
		create_character = {
			first_name = Gulbahar
			birth_location = root.capital
			age = 16
			culture = root.culture
			religion = root.religion
			female = yes
			fertility = 100
			adm = 100
			dip = 100
			mil = 100
			add_character_modifier = {
				modifier = banned_from_cabinet
				years = modifier_duration_permanent
				mode = add_and_extend
			}
			marry_character = character:dbo_fatih_sultan_mehmed
		}
	}
	option = {
		name = dbo_dhe_sultans_fatih_sultan_mehmed.3.a
		historical_option = yes
		if = {
			limit = {
				character:dbo_fatih_sultan_mehmed = {
					is_ruler = no
				}
			}
			set_new_ruler = character:dbo_fatih_sultan_mehmed
		}
		# dbo_fatih_sultan_mehmed and Orban Events
		custom_tooltip = {
			text = dbo_fatih_sultan_mehmed_and_orban_tt
			trigger_event_silently = {
				id = dbo_dhe_sultans_fatih_sultan_mehmed.11
				months = 3
			}
		}
		dbo_event_mild_bonus = yes
	}
}
