dbo_location_is_homeland = {
	owns = $location$
	$location$ = {
		is_core_of = root
	}
}

dbo_in_circulation_new_world_goods_trigger = {
	any_market_present_in_country = {
		OR = {
			is_produced_in_market = $goods$
			is_traded_in_market = $goods$
		}
	}
}

dbo_rgo_replaced_goods = {
	OR = {
		root = goods:fish
		root = goods:clay
		root = goods:sand
		root = goods:wool
	}
}

dbo_turkish_pashas_rebel_allegiance_trigger = {
	OR = {
		pop_type = pop_type:soldiers
		AND = {
			OR = {
				estate_type = estate_type:nobles_estate
				estate_type = estate_type:peasants_estate
			}
			culture = {
				has_culture_group = culture_group:turkic_group
			}
		}
	}
}
