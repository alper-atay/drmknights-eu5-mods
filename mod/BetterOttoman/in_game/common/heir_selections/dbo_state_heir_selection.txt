dbo_state_heir_selection = {
	traverse_family_tree = no	# do we use the family tree to calculate who's in line
	include_ruler_siblings = yes
	allow_male = yes
	ignore_ruler = yes
	cached = yes
	allow_foreign_ruler = no
	potential = {
		government_type = government_type:dbo_government_type_state
	}
	locked = {
		government_type = government_type:dbo_government_type_state
	}
	include_other_countries = {
		scope:target = {
			is_subject_of = root
		}
	}
	candidate_country = {
		scope:target = {
			is_subject_of = root
		}
	}
	heir_is_allowed = {
		# extra per-candidate filters if you ever want them
	}
	succession_effect = {
		add_legitimacy = 25
	}
	# candidates may only have these Estate types. If left blank it will work for all estates.
	allowed_estates = {
		nobles_estate
	}
	calc = {
		if = {
			limit = {
				exists = root
			}
			add = {
				desc = "HEIR_SELECTION_OLDER_HEIR_DESC"
				value = root.character_age
				multiply = 0.25
			}
			add = {
				desc = "HEIR_SELECTION_ADM_DESC"
				value = root.adm
				multiply = 2
			}
			add = {
				desc = "HEIR_SELECTION_DIP_DESC"
				value = root.dip
			}
			add = {
				desc = "HEIR_SELECTION_MIL_DESC"
				value = root.mil
				multiply = 0.75
			}
			if = {
				limit = {
					exists = root.father
				}
				if = {
					limit = {
						any_parent = {
							this = scope:target.ruler
						}
					}
					add = {
						desc = "HEIR_SELECTION_PARENT_IS_CURRENT_RULER_DESC"
						value = 75						# your tuned value
					}
				}
			}
			if = {
				limit = {
					OR = {
						is_general = yes
						is_admiral = yes
					}
				}
				add = {
					desc = "HEIR_SELECTION_GENERAL_OR_ADMIRAL_DESC"
					value = 25					# your tuned value
				}
			}
			if = {
				limit = {
					OR = {
						has_trait_category = general
						has_trait_category = admiral
					}
				}
				add = {
					desc = "HEIR_SELECTION_GENERAL_OR_ADMIRAL_TRAIT_DESC"
					value = 25					# your tuned value
				}
			}
			if = {
				limit = {
					modifier:blocked_from_being_ruler = yes
				}
				add = {
					desc = "HEIR_SELECTION_IS_UNSUITED_TO_RULE_DESC"
					value = -1000
				}
			}
		}
	}
}
