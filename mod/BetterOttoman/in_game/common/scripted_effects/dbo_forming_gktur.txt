dbo_forming_gktur_religion = {
	change_religion = religion:tengri
	change_religion_for_ruler_and_family = {
		country = root
		religion = religion:tengri
	}
	set_court_language = culture.dialect
	set_liturgical_language = culture.dialect
}

dbo_forming_gktur_religion_every_turkic_group_change_to_tengri = {
	every_country = {
		every_pop = {
			limit = {
				NOT = {
					religion = religion:tengri
				}
				culture = {
					has_culture_group = culture_group:turkic_group
				}
			}
			split_pop = {
				fraction = 0.5
				religion = religion:tengri
			}
		}
	}
}

dbo_forming_gktur_government_reforms = {
	hidden_effect = {
		lock_government_reform_effect = {
			type = anatolian_beylik
		}
		lock_government_reform_effect = {
			type = dbo_destiny_cultural_capital_constantinople
		}
		lock_government_reform_effect = {
			type = dbo_destiny_legacy_of_seljuks
		}
		lock_government_reform_effect = {
			type = dbo_destiny_ottoman_empire
		}
		if = {
			limit = {
				has_reform = government_reform:anatolian_beylik
			}
			remove_reform = government_reform:anatolian_beylik
		}
		if = {
			limit = {
				has_reform = government_reform:dbo_destiny_cultural_capital_constantinople
			}
			remove_reform = government_reform:dbo_destiny_cultural_capital_constantinople
		}
		if = {
			limit = {
				has_reform = government_reform:dbo_destiny_legacy_of_seljuks
			}
			remove_reform = government_reform:dbo_destiny_legacy_of_seljuks
		}
		if = {
			limit = {
				has_reform = government_reform:dbo_destiny_ottoman_empire
			}
			remove_reform = government_reform:dbo_destiny_ottoman_empire
		}
		unlock_government_reform_effect = {
			type = dbo_destiny_children_of_tengri
		}
		unlock_government_reform_effect = {
			type = dbo_destiny_wing_of_turk
		}
	}
	add_reform = government_reform:dbo_destiny_children_of_tengri
	add_reform = government_reform:dbo_destiny_wing_of_turk
}

dbo_forming_gktur_buildings = {
	unlock_building_effect = {
		type = dbo_otag
	}
	unlock_building_effect = {
		type = dbo_ulu_otag
	}
	custom_tooltip = {
		text = dbo_forming_gktur_every_owned_location_construct_otag_tt
		every_owned_location = {
			if = {
				limit = {
					can_build_building = building_type:dbo_otag
				}
				construct_building = {
					building_type = building_type:dbo_otag
					cost_multiplier = 0
					cost_multiplier_reason = "otag_must_be_free"
					instant = yes
				}
			}
		}
	}
}

dbo_forming_gktur_buildings_destroy_religious_category = {
	every_owned_location = {
		every_buildings_in_location = {
			limit = {
				NOT = {
					building_type = building_type:temple
				}
				building_category = religious_category
				building_type = {
					is_special_building = no
				}
			}
			location = {
				destroy_building = prev
			}
		}
	}
}

dbo_forming_gktur_location_add_core = {
	custom_tooltip = {
		text = dbo_forming_gktur_historical_asia_locations_add_core_tt
		every_location_in_the_world = {
			limit = {
				is_ownable = yes
				OR = {
					region = region:anatolia_region
					region = region:caucasus_region
					region = region:steppes_region
					region = region:ural_region
					region = region:khorasan_region
					region = region:xinjiang_region
					region = region:mongolia_region
					region = region:east_siberia_region
					region = region:west_siberia_region
				}
			}
			add_core = root
		}
	}
}

dbo_forming_gktur_harmonize_culture = {
	every_culture = {
		limit = {
			NOT = {
				this = root.culture
			}
			has_culture_group = culture_group:turkic_group
		}
		save_temporary_scope_as = target_culture
		root = {
			culture = {
				set_cultural_view = {
					target = scope:target_culture
					value = kindred
				}
				reverse_set_cultural_view = {
					target = scope:target_culture
					value = kindred
				}
			}
			if = {
				limit = {
					culture_percentage_in_country = {
						culture = scope:target_culture
						value >= 0.01
					}
				}
				add_accepted_culture = scope:target_culture
			}
		}
	}
}
