dcma_marry = {
	every_character = {
		limit = {
			is_female = no
			is_alive = yes
			is_adult = yes
			has_available_marriage_slot = yes
			num_of_spouses < 2
			age_in_years < 40
			has_estate = estate_type:crown_estate
			total_abilities > 120
		}
		save_scope_as = target_crown_male
		if = {
			limit = {
				exists = scope:target_crown_male
			}
			owner = {
				create_character = {
					female = yes
					religion = scope:target_crown_male.religion
					culture = scope:target_crown_male.culture
					random_estate_type_trigger = {
						NOR = {
							this = estate_type:nobles_estate
							this = estate_type:clergy_estate
							this = estate_type:dhimmi_estate
						}
					}
					adm = {
						90
						100
					}
					dip = {
						90
						100
					}
					mil = {
						90
						100
					}
					age = 16
					save_scope_as = target_created_lowborn_female
				}
				if = {
					limit = {
						exists = scope:target_created_lowborn_female
					}
					scope:target_crown_male ?= {
						marry_character = scope:target_created_lowborn_female
					}
				}
			}
		}
	}
}
