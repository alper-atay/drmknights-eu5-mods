dbrgo_replace_fish_with_best_option = {
	dbrgo_build_fishing_village_is_coastal_location = yes
	if = {
		limit = {
			continent = continent:europe
		}
		random_list = {
			20 = {
				change_raw_material = goods:wheat
			}
			20 = {
				change_raw_material = goods:rice
			}
			20 = {
				change_raw_material = goods:millet
			}
			20 = {
				change_raw_material = goods:livestock
			}
			15 = {
				change_raw_material = goods:legumes
			}
			15 = {
				change_raw_material = goods:fruit
			}
			10 = {
				change_raw_material = goods:wild_game
			}
		}
	}
}

dbrgo_build_fishing_village_is_coastal_location = {
	if = {
		limit = {
			has_owner = yes
			is_coastal = yes
			can_build_building = building_type:fishing_village
		}
		change_building_level_in_location = {
			building = building_type:fishing_village
			value = 1
		}
	}
}
