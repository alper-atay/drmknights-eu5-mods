			25 = {
				change_raw_material = goods:coal
			}
			20 = {
				change_raw_material = goods:iron
			}
			20 = {
				change_raw_material = goods:copper
			}
			20 = {
				change_raw_material = goods:tin
			}
			15 = {
				change_raw_material = goods:lead
			}
			10 = {
				change_raw_material = goods:gems
			}
			10 = {
				change_raw_material = goods:silver
			}
			5 = {
				change_raw_material = goods:goods_gold
			}
		}
	}
	else_if = {
		limit = {
			topography = hills
		}
		random_list = {
			25 = {
				change_raw_material = goods:coal
			}
			20 = {
				change_raw_material = goods:iron
			}
			20 = {
				change_raw_material = goods:copper
			}

				change_raw_material = goods:tin
			}
		}
	}

			limit = {
				region = region:east_coast_region
			}
			dbrgo_best_rgo_for_east_coast_region = yes
		}
		else_if = {
		}
	}
}

		}
